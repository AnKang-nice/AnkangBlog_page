react render ​

对于react而言，性能优化主要分两大部分：

一部分是前端框架通用性能优化，比如像组件按需加载（路由懒加载、懒渲染、虚拟列表...）、配置分包、CDN方式引入库等；

另一部分是针对react渲染的性能优化方法 -- 核心思想：减少不必要的组件更新；

本次主要介绍的是函数组件部分优化方式，在此之前我想先介绍下react的更新策略

react基础 ​

react架构

scheduler阶段（调度器）：分发优先级更高的任务

render阶段（协调器）：找到哪些节点发生了变化，并且给相应fiber打上标签

commit阶段（渲染器）：根据打好标签的节点处理渲染到视图，再执行相应dom操作或部分生命周期

react render简单流程 ​

render阶段的核心是为了创建fiber树；fiber树由遍历fiber节点生成；fiber节点是由方法通过reactElement创建的一个数据结构;

本次以一段简单代码为示例去简单的分析下简易的render流程 ​

遍历生成fiber树的两个核心方法 ​

beginWork - 追寻 ​

作用：决定节点是否复用，返回下个节点；

javascript
function beginWork(
  current: Fiber | null,
  workInProgress: Fiber,
  renderLanes: Lanes,
): Fiber | null {
  if (__DEV__) {
    if (workInProgress._debugNeedsRemount && current !== null) {
      // This will restart the begin phase with a new fiber.
      return remountFiber(
        current,
        workInProgress,
        createFiberFromTypeAndProps(
          workInProgress.type,
          workInProgress.key,
          workInProgress.pendingProps,
          workInProgress._debugOwner || null,
          workInProgress.mode,
          workInProgress.lanes,
        ),
      );
    }
  }
	// didReceiveUpdate  表示是否有新的props更新，有则会设置为true，没有则是false
  if (current !== null) {
    // 界定是创建还是更新
    const oldProps = current.memoizedProps;
    const newProps = workInProgress.pendingProps;

    const ifAllow = oldProps !== newProps; // 验证判断
    
    // 检查props context是否发生改变
    if (
      ifAllow || // 全等比较
      hasLegacyContextChanged() || // legacy模式
      // Force a re-render if the implementation changed due to hot reload:
      (__DEV__ ? workInProgress.type !== current.type : false)
    ) {
      // 表示props   context  发生了改变    需要进行更新
      didReceiveUpdate = true; 
    } else {
      // checkScheduledUpdateOrContext函数检查当前fiber节点上的lanes是否存在于renderLanes中
      // 存在则说明当前fiber节点需要更新，不存在则不需要更新则复用之前的节点
      const hasScheduledUpdateOrContext = checkScheduledUpdateOrContext(
        current,
        renderLanes,
      );
      if (
        !hasScheduledUpdateOrContext &&
        (workInProgress.flags & DidCapture) === NoFlags
      ) {
        // 当前节点没有pending  update   进行bailout过程  - 可复用
        didReceiveUpdate = false;
        // 复用filber节点  返回的是workInProgress.child ｜ null
        // 返回的是null	  表示当前节点及其子节点都复用   beginWork停止
        // 返回的是workInProgress.child   表示当前节点复用    继续执行beginWork
        return attemptEarlyBailoutIfNoScheduledUpdate(
          current,
          workInProgress,
          renderLanes,
        );
      }
       if ((current.flags & ForceUpdateForLegacySuspense) !== NoFlags) {
        didReceiveUpdate = true;
      } else {
        didReceiveUpdate = false;
      }
    }
  } else {
    //  创建
  }
	// *******************************  分割
  workInProgress.lanes = NoLanes;

  switch (workInProgress.tag) {// 表示组件类型
    // 函数组件
    case FunctionComponent: {
      const Component = workInProgress.type;
      const unresolvedProps = workInProgress.pendingProps;
      const resolvedProps =
        workInProgress.elementType === Component
          ? unresolvedProps
          : resolveDefaultProps(Component, unresolvedProps);
      // 简单的理解  当前函数组件在渲染过程中有更新   创建了新reactElement   -->   fiber节点
      // 对于引用类型props   那么肯定不===
      return updateFunctionComponent(
        current,
        workInProgress,
        Component,
        resolvedProps,
        renderLanes,
      );
    }
    // 根组件
    case HostRoot:
      return updateHostRoot(current, workInProgress, renderLanes);
    // 元素  div ...
    case HostComponent:
      return updateHostComponent(current, workInProgress, renderLanes);
    case HostText:
      return updateHostText(current, workInProgress);
    case MemoComponent: {
      const type = workInProgress.type;
      const unresolvedProps = workInProgress.pendingProps;
      // Resolve outer props first, then resolve inner props.
      let resolvedProps = resolveDefaultProps(type, unresolvedProps);
      if (__DEV__) {
        if (workInProgress.type !== workInProgress.elementType) {
          const outerPropTypes = type.propTypes;
          if (outerPropTypes) {
            checkPropTypes(
              outerPropTypes,
              resolvedProps, // Resolved for outer only
              'prop',
              getComponentNameFromType(type),
            );
          }
        }
      }
      resolvedProps = resolveDefaultProps(type.type, resolvedProps);
      return updateMemoComponent(
        current,
        workInProgress,
        type,
        resolvedProps,
        renderLanes,
      );
    }
    case SimpleMemoComponent: {
      return updateSimpleMemoComponent(
        current,
        workInProgress,
        workInProgress.type,
        workInProgress.pendingProps,
        renderLanes,
      );
    }
    ...
  }

  throw new Error(
    `Unknown unit of work tag (${workInProgress.tag}). This error is likely caused by a bug in ` +
      'React. Please file an issue.',
  );
}

结束条件：

一次 beginWork 会不断的访问子节点，其停止条件为:

  1. 访问到的节点不存在子节点
  2. 访问到的节点只有一个纯文本节点，例如+1，访问到 input 对应的 fiber 就会结束本次 beginWork
  3. 该 fiber 本次不需要更新，其所有的子 fiber 也不需要更新，则会提前结束 beginWork阶段。在 react 中这种情况也称为 bailout。
​

作用：简单理解：找当前节点的下个同级节点，在无同级节点情况下，回溯到父级节点，直到为null

结束条件：

当遍历到某个节点时 beginWork 结束，会从该节点开始进行 completeWork 阶段。completeWork 阶段结束条件如下:

  1. 当前节点存在兄弟节点
  2. 当前节点不存在兄弟节点，但是其祖先节点存在兄弟节点
  3. 当前节点不存在兄弟节点，也不存在父节点

其中情况 1,2 会以得到的兄弟节点为起点，开始新一轮的 beginWork 阶段。情况 3 表示该 fiber 树已经遍历完成，整个协调阶段也会结束。

现象： ​

如示例Demo1，当触发add事件时，num发生变化，而Child1虽然跟num没有关系，但是也重新渲染了

目标： ​

不让Child1渲染

原因： ​

数据更新后， render阶段，oldprops{} !== newProps{} ，didReceiveUpdate=true，会执行updateFunctionComponent - 重新更新（render）

思路： ​

让child1组件在创建fiber树的时候，oldprops{} === newProps{}，进行复用

方案： ​

注释： ​

fiber节点

理解为一种数据结构，保存着dom相关信息、fiber树相关引用等，fiber节点的更新等一系列操作都是由其中的属性决定的

javascript
export type Fiber = {|
  // 作为静态数据结构，存储节点 dom 相关信息
  tag: WorkTag, // 节点的类型，取决于 react 的元素类型
  key: null | string,
  elementType: any, // 元素类型
  type: any, // 定义与此fiber关联的功能或类。对于组件，它指向构造函数；对于DOM元素，它指定HTML tag
  stateNode: any, // 真实 dom 节点
  
  // fiber 链表树相关
  return: Fiber | null, // 父 fiber
  child: Fiber | null, // 第一个子 fiber
  sibling: Fiber | null, // 下一个兄弟 fiber
  index: number, // 在父 fiber 下面的子 fiber 中的下标
  
  ref:
    | null
    | (((handle: mixed) => void) & {_stringRef: ?string, ...})
    | RefObject,
    
  // 工作单元，用于计算 state 和 props 渲染
  pendingProps: any, // 本次渲染需要使用的 props
  memoizedProps: any, // 上次渲染使用的 props
  updateQueue: mixed, // 用于状态更新、回调函数、DOM更新的队列
  memoizedState: any, // 上次渲染后的 state 状态
  dependencies: Dependencies | null, // contexts、events 等依赖
  
  mode: TypeOfMode,
  
  // 副作用相关
  flags: Flags, // 记录更新时当前 fiber 的副作用(删除、更新、替换等)状态
  subtreeFlags: Flags, // 当前子树的副作用状态
  deletions: Array<Fiber> | null, // 要删除的子 fiber
  nextEffect: Fiber | null, // 下一个有副作用的 fiber
  firstEffect: Fiber | null, // 指向第一个有副作用的 fiber
  lastEffect: Fiber | null, // 指向最后一个有副作用的 fiber 
  
  // 优先级相关
  lanes: Lanes,
  childLanes: Lanes,
  
  alternate: Fiber | null, // 指向 workInProgress fiber 树中对应的节点
  
  actualDuration?: number,
  actualStartTime?: number,
  selfBaseDuration?: number,
  treeBaseDuration?: number,
  _debugID?: number,
  _debugSource?: Source | null,
  _debugOwner?: Fiber | null,
  _debugIsCurrentlyTiming?: boolean,
  _debugNeedsRemount?: boolean,
  _debugHookTypes?: Array<HookType> | null,
|};
